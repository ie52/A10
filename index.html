package com.example; import javax.servlet.annotation.WebServlet; import javax.servlet.http.Cookie; import javax.servlet.http.HttpServlet; import javax.servlet.http.HttpServletRequest; import javax.servlet.http.HttpServletResponse; import java.io.IOException; @WebServlet("/") public class CookieServlet extends HttpServlet { protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException { // Check if cookie exists String cookieValue = null; Cookie[] cookies = request.getCookies(); if (cookies != null) { for (Cookie cookie: cookies) { if ("user".equals(cookie.getName())) { cookieValue = cookie.getValue(); break; } } } // Simple HTML page with cookie info and a button response.setContentType("text/html"); response.getWriter().println(""); response.getWriter().println("Cookie Demo"); response.getWriter().println(""); response.getWriter().println("

Cookie Demo

"); if (cookieValue != null) { response.getWriter().println("

Cookie value: " + cookieValue + "

"); } else { response.getWriter().println("

No cookie found

"); } response.getWriter().println("
"); response.getWriter().println(""); response.getWriter().println("
"); response.getWriter().println(""); } protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException { // Create a cookie that lasts for 1 day Cookie cookie = new Cookie("user", "testUser"); cookie.setMaxAge(24 * 60 * 60); response.addCookie(cookie); // Redirect back to the main page response.sendRedirect(request.getContextPath() + "/"); } }
